+
+    // Shared reference: onTouch writes here, ForestVisual observes via @ObservedObject.
+    // Created once per ForestScene value; struct copy semantics preserve the reference.
+    private let touchState = ForestTouchState()
+
+    private let logger = Logger(subsystem: "app.nagi", category: "ForestScene")
+
+    @ViewBuilder
+    func makeVisual(context: SensoryContext) -> some View {
+        ForestVisual(touchState: touchState)
+    }
+
+    func prepare(context: SensoryContext) async {
+        // Register haptic patterns (leaf crunch / branch creak / sustained rustle)
+        do {
+            let leafCrunch = try HapticEngine.makeForestLeafCrunchPattern()
+            context.haptics.register(name: "leaf_crunch", pattern: leafCrunch)
+        } catch {
+            logger.warning("leaf_crunch pattern failed: \(error.localizedDescription, privacy: .public)")
+        }
+
+        do {
+            let branchCreak = try HapticEngine.makeForestBranchCreakPattern()
+            context.haptics.register(name: "branch_creak", pattern: branchCreak)
+        } catch {
+            logger.warning("branch_creak pattern failed: \(error.localizedDescription, privacy: .public)")
+        }
+
+        do {
+            let leafRustle = try HapticEngine.makeForestLeafRustlePattern()
+            context.haptics.register(name: "leaf_rustle", pattern: leafRustle)
+        } catch {
+            logger.warning("leaf_rustle pattern failed: \(error.localizedDescription, privacy: .public)")
+        }
+
+        // Register audio asset
+        guard let url = Bundle.main.url(forResource: "forest_ambient", withExtension: "wav")
+                ?? Bundle.main.url(forResource: "forest_ambient", withExtension: "aiff") else {
+            logger.warning("forest_ambient not found in bundle — skipping audio registration")
+            return
+        }
+        context.audio.registerSoundAsset(identifier: "forest_ambient", url: url)
+    }
+
+    func onTouch(_ event: TouchEvent, context: SensoryContext) {
+        switch event {
+        case .began(let position):
+            let eventID = context.clock.beginEvent(label: "forest.tap")
+            touchState.previousTouchLocation = nil
+            touchState.touchLocation = position
+            touchState.isTouching = true
+            // Initial step on the forest floor: a crisp leaf crunch.
+            context.haptics.play(name: "leaf_crunch")
+            context.clock.completeEvent(id: eventID, actuator: "haptic")
+
+        case .moved(let from, let to, _):
+            touchState.previousTouchLocation = from
+            touchState.touchLocation = to
+            touchState.isTouching = true
+
+        case .ended(let position):
+            touchState.isTouching = false
+            touchState.touchLocation = position
+            // Lift-off: leaves settle back into a brief rustle.
+            context.haptics.play(name: "leaf_rustle")
+
+        case .longPress(let position):
+            // Holding ground evokes a deeper, slower branch creak.
+            touchState.touchLocation = position
+            context.haptics.play(name: "branch_creak")
+        }
+    }
+
+    func teardown(context: SensoryContext) {
+        // Engine lifecycle is owned by SceneDirector; nothing to do here.
+    }
+}

+#include <metal_stdlib>
+using namespace metal;

+// Mirrors the layout in FluidShaders.metal — keep these fields in sync.
+struct FluidUniforms {
+    float2 resolution;
+    float2 touch;
+    float2 touchVelocity;
+    float touchActive;
+    float dt;
+    float viscosity;
+    float diffusion;
+    float time;
+    float3 tint;
+};
+
+// MARK: - Hash / noise utilities (prefix fr_ to avoid name collisions across shader files)
+
+static float fr_hash(float2 p) {
+    float3 p3 = fract(float3(p.xyx) * 0.1031);
+    p3 += dot(p3, p3.yzx + 33.33);
+    return fract((p3.x + p3.y) * p3.z);
+}
+
+static float fr_hash1(float n) {
+    return fract(sin(n) * 43758.5453123);
+}
+
+static float fr_noise(float2 p) {
+    float2 i = floor(p);
+    float2 f = fract(p);
+    f = f * f * (3.0 - 2.0 * f);
+    float a = fr_hash(i);
+    float b = fr_hash(i + float2(1, 0));
+    float c = fr_hash(i + float2(0, 1));
+    float d = fr_hash(i + float2(1, 1));
+    return mix(mix(a, b, f.x), mix(c, d, f.x), f.y);
+}
+
+// FBM with subtle drift — used for foliage shimmer.
+static float fr_fbm(float2 p, float time, int octaves) {
+    float v = 0.0;
+    float amp = 0.5;
+    float2 shift = float2(time * 0.18, time * -0.12);
+    for (int i = 0; i < octaves; i++) {
+        v += amp * fr_noise(p + shift);
+        p *= 2.07;
+        amp *= 0.5;
+        shift *= 1.15;
+    }
+    return v;
+}
+
+// Slow FBM for atmospheric haze drifting between trunks.
+static float fr_fbm_haze(float2 p, float time) {
+    float v = 0.0;
+    float amp = 0.5;
+    float2 shift = float2(time * 0.04, time * 0.02);
+    for (int i = 0; i < 4; i++) {
+        v += amp * fr_noise(p + shift);
+        p *= 1.95;
+        amp *= 0.5;
+    }
+    return v;
+}
+
+// MARK: - Bark grain (per-trunk procedural texture)
+
+static float fr_bark(float2 uv, float seed) {
+    float vertical = fr_noise(uv * float2(1.5, 22.0) + seed);
+    float cracks = fr_noise(uv * float2(8.0, 60.0) + seed * 1.7);
+    float knots = smoothstep(0.6, 0.85, fr_noise(uv * float2(2.5, 5.0) + seed * 3.1));
+    return mix(vertical * 0.7 + cracks * 0.3, knots, 0.25);
+}

+// MARK: - forest_render kernel
+
+kernel void forest_render(
+    texture2d<float, access::write> output [[texture(0)]],
+    constant FluidUniforms &u             [[buffer(0)]],
+    uint2 gid                             [[thread_position_in_grid]]
+) {
+    int w = int(u.resolution.x);
+    int h = int(u.resolution.y);
+    if (int(gid.x) >= w || int(gid.y) >= h) return;
+
+    float2 uv = float2(gid) / u.resolution;
+    // Y-flip: 0 = bottom (forest floor), 1 = top (canopy).
+    float2 fUV = float2(uv.x, 1.0 - uv.y);
+
+    float t = u.time;
+
+    // --- Touch UV in flipped space ---
+    float2 touchUV = float2(u.touch.x / u.resolution.x, 1.0 - u.touch.y / u.resolution.y);
+    float touchActive = (u.touchActive > 0.5) ? 1.0 : 0.0;
+
+    // ---- Background: deep forest gradient ----
+    // Floor (mossy umber) → mid (deep green-black) → canopy (faint blue-green sky)
+    float3 floorColor   = float3(0.04, 0.06, 0.04);
+    float3 deepColor    = float3(0.02, 0.05, 0.04);
+    float3 canopyColor  = float3(0.04, 0.10, 0.09);
+    float3 skyHint      = float3(0.10, 0.18, 0.16);
+
+    float3 bg;
+    if (fUV.y < 0.35) {
+        bg = mix(floorColor, deepColor, fUV.y / 0.35);
+    } else if (fUV.y < 0.85) {
+        bg = mix(deepColor, canopyColor, (fUV.y - 0.35) / 0.5);
+    } else {
+        bg = mix(canopyColor, skyHint, (fUV.y - 0.85) / 0.15);
+    }
+
+    float3 color = bg;
+
+    // ---- Atmospheric haze / mist drifting between trunks ----
+    {
+        float2 hazeUV = fUV * float2(2.5, 1.4) + float2(0.0, t * 0.02);
+        float haze = fr_fbm_haze(hazeUV, t);
+        float hazeMask = smoothstep(0.45, 0.85, haze) * smoothstep(0.0, 0.4, fUV.y) * 0.18;
+        color = mix(color, float3(0.18, 0.22, 0.20), hazeMask);
+    }
+
+    // ---- Tree trunks (vertical slabs across the width) ----
+    // Several deep, ancient trunks at different x positions / widths / sway.
+    const int TRUNK_COUNT = 6;
+    float trunkXs[6]      = { 0.07, 0.22, 0.38, 0.56, 0.74, 0.91 };
+    float trunkWidths[6]  = { 0.045, 0.075, 0.055, 0.090, 0.060, 0.050 };
+    float trunkSeeds[6]   = { 11.3, 27.9, 5.4, 41.2, 17.8, 33.5 };
+    // Subtle parallax / depth tone — back trunks sit cooler & darker.
+    float trunkDepth[6]   = { 0.85, 0.55, 1.00, 0.40, 0.75, 0.90 };
+
+    for (int i = 0; i < TRUNK_COUNT; i++) {
+        float cx = trunkXs[i];
+        float halfWidth = trunkWidths[i];
+        // Very gentle natural lean (different for each trunk).
+        float lean = sin(t * 0.18 + trunkSeeds[i]) * 0.004;
+        float dx = (fUV.x - cx) - lean * fUV.y;
+
+        // Soft trunk silhouette.
+        float trunkMask = smoothstep(halfWidth, halfWidth * 0.7, abs(dx));
+        if (trunkMask <= 0.001) continue;
+
+        // Bark texture mapped to local trunk UV.
+        float2 barkUV = float2((dx / halfWidth) * 0.5 + 0.5, fUV.y);
+        float bark = fr_bark(barkUV, trunkSeeds[i]);
+
+        // Base trunk color: dark mossy bark with depth-tinted variance.
+        float3 barkBase = mix(float3(0.05, 0.04, 0.03),
+                              float3(0.13, 0.10, 0.06),
+                              bark);
+        // Push back trunks toward green-black for atmospheric depth.
+        barkBase = mix(barkBase * 0.45, barkBase, trunkDepth[i]);
+
+        // Faint moss patches near the base (lower 30%).
+        float mossy = fr_noise(barkUV * float2(4.0, 6.0) + trunkSeeds[i]);
+        mossy = smoothstep(0.55, 0.85, mossy) * smoothstep(0.35, 0.0, fUV.y);
+        barkBase = mix(barkBase, float3(0.08, 0.18, 0.06), mossy * 0.55);
+
+        color = mix(color, barkBase, trunkMask);
+    }
+
+    // ---- Canopy / leaf shimmer (top 55%) ----
+    if (fUV.y > 0.30) {
+        float canopyT = (fUV.y - 0.30) / 0.70;
+        // Two layers of FBM at different scales — leaves nearby and far.
+        float leavesNear = fr_fbm(fUV * float2(8.0, 6.0) + float2(0.0, -t * 0.05), t, 4);
+        float leavesFar  = fr_fbm(fUV * float2(20.0, 16.0) + float2(t * 0.04, 0.0), t, 3);
+        float foliage = mix(leavesNear, leavesFar, 0.55);
+
+        // Mask leaves to canopy region with soft falloff.
+        float canopyMask = smoothstep(0.0, 0.35, canopyT) * smoothstep(1.05, 0.75, canopyT);
+
+        // Wind-driven shimmer: small breeze ripple over the foliage.
+        float breeze = sin(fUV.x * 18.0 + t * 1.4) * 0.5 + 0.5;
+        breeze = mix(0.6, 1.0, breeze * smoothstep(0.4, 0.9, foliage));
+
+        // Leaf color: mossy green deepening into blue-green canopy.
+        float3 leafDark = float3(0.03, 0.08, 0.05);
+        float3 leafMid  = float3(0.07, 0.20, 0.10);
+        float3 leafHigh = float3(0.18, 0.36, 0.20);
+        float3 leafColor = mix(leafDark, leafMid, foliage);
+        leafColor = mix(leafColor, leafHigh, smoothstep(0.65, 0.95, foliage) * 0.6);
+        leafColor *= breeze;
+
+        // Apply foliage as a soft overlay over the canopy region.
+        color = mix(color, leafColor, canopyMask * smoothstep(0.25, 0.6, foliage));
+    }
+
+    // ---- God rays / 木漏れ日 (light shafts piercing the canopy) ----
+    // 5 narrow shafts at fixed positions, slowly breathing in/out.
+    {
+        const int SHAFT_COUNT = 5;
+        float shaftXs[5]    = { 0.18, 0.34, 0.50, 0.68, 0.83 };
+        float shaftAngles[5] = { 0.06, -0.04, 0.02, -0.07, 0.05 };
+        float shaftSeeds[5]  = { 2.1, 4.7, 1.3, 6.2, 8.9 };
+
+        for (int i = 0; i < SHAFT_COUNT; i++) {
+            // Light originates above the canopy and angles down.
+            float angle = shaftAngles[i];
+            // Distance from the shaft's center line.
+            float lineX = shaftXs[i] + (1.0 - fUV.y) * angle;
+            float distToShaft = abs(fUV.x - lineX);
+
+            // Narrow shaft — sharper near the top, wider near the floor.
+            float shaftWidth = 0.012 + (1.0 - fUV.y) * 0.018;
+            float shaftMask = smoothstep(shaftWidth, 0.0, distToShaft);
+
+            // Vertical fade: brightest in the upper canopy, gentle at the floor.
+            float verticalFade = smoothstep(0.05, 0.55, fUV.y) * smoothstep(1.05, 0.45, fUV.y);
+
+            // Slow breathing pulse (different phase per shaft).
+            float pulse = sin(t * 0.45 + shaftSeeds[i]) * 0.25 + 0.75;
+
+            // Atmospheric noise grain so the shafts feel like light through dust.
+            float dust = fr_noise(fUV * float2(60.0, 30.0) + float2(shaftSeeds[i] * 5.0, t * 0.1));
+            float dustMask = mix(0.7, 1.0, dust);
+
+            float shaftIntensity = shaftMask * verticalFade * pulse * dustMask;
+
+            // Warm diffuse light color — warm sunlight tinted by green canopy.
+            float3 shaftColor = float3(1.0, 0.92, 0.62);
+            color += shaftColor * shaftIntensity * 0.55;
+

+            }
+        }
+    }
+
+    // ---- Drifting motes / pollen catching the light ----
+    {
+        for (int i = 0; i < 18; i++) {
+            float seed = float(i) * 91.337;
+            float bx = fr_hash1(seed) * 0.95 + 0.025;
+            float by = fr_hash1(seed + 1.0);
+            float speed = fr_hash1(seed + 2.0) * 0.06 + 0.02;
+            float drift = fr_hash1(seed + 3.0) * 2.0 - 1.0;
+
+            // Slow downward drift with horizontal swing.
+            float py = fract(by - t * speed);
+            float px = bx + sin(t * 0.4 + seed) * 0.012 * drift;
+
+            float2 motePos = float2(px, py);
+            float dist = length((fUV - motePos) * float2(1.0, u.resolution.y / u.resolution.x));
+            float moteSize = 0.0025 + fr_hash1(seed + 4.0) * 0.0025;
+            float moteGlow = exp(-dist * dist / (moteSize * moteSize));
+
+            // Motes only visible when crossing a god-ray (x near a shaft).
+            float visibility = 0.25;
+            float shaftXsLocal[5] = { 0.18, 0.34, 0.50, 0.68, 0.83 };
+            for (int j = 0; j < 5; j++) {
+                float shaftAngle = (j == 0) ? 0.06 : (j == 1) ? -0.04 : (j == 2) ? 0.02 : (j == 3) ? -0.07 : 0.05;
+                float lx = shaftXsLocal[j] + (1.0 - py) * shaftAngle;
+                visibility = max(visibility, exp(-pow((px - lx) / 0.05, 2.0)));
+            }
+
+            color += float3(1.0, 0.88, 0.55) * moteGlow * visibility * 0.55;
+        }
+    }
+
+    // ---- Touch interaction: leaves rustle / light blooms at touch ----
+    if (touchActive > 0.5) {
+        float dist = distance(fUV, touchUV);
+        // Soft warm halo where the finger "disturbs" the foliage.
+        float touchHalo = exp(-dist * dist / 0.012);
+        color += float3(0.55, 0.45, 0.20) * touchHalo * 0.35;
+
+        // Leaf agitation: extra shimmer near touch in the canopy region.
+        if (fUV.y > 0.30) {
+            float agitate = fr_fbm(fUV * 24.0 + t * 3.0, t, 3);
+            float ring = smoothstep(0.10, 0.04, dist) * smoothstep(0.0, 0.04, dist);
+            color += float3(0.10, 0.22, 0.10) * agitate * ring * 1.5;
+        }
+    }
+
+    // ---- Forest floor: hint of leaf litter (bottom 15%) ----
+    if (fUV.y < 0.15) {
+        float litter = fr_fbm(fUV * float2(40.0, 18.0), 0.0, 3);
+        float litterMask = smoothstep(0.0, 0.15, fUV.y) * smoothstep(0.15, 0.05, fUV.y);
+        // Avoid double-painting where trunks already drew.
+        float3 leafLitter = mix(float3(0.06, 0.04, 0.02),
+                                float3(0.14, 0.10, 0.04),
+                                litter);
+        color = mix(color, leafLitter, litterMask * 0.5);
+    }
+
+    // ---- Vignette ----
+    float2 vigUV = uv - 0.5;
+    float vignette = 1.0 - dot(vigUV, vigUV) * 1.1;
+    color *= saturate(vignette);
+
+    // ---- ACES tonemap (preserves highlights in light shafts) ----
+    color *= 1.35;
+    float3 x = color;
+    color = (x * (2.51 * x + 0.03)) / (x * (2.43 * x + 0.59) + 0.14);
+
+    // Gamma
+    color = pow(saturate(color), float3(1.0 / 2.2));
+
+    // Tint
+    color *= u.tint;
+
+    output.write(float4(saturate(color), 1.0), gid);
+}
