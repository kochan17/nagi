#include <metal_stdlib>
using namespace metal;

struct FluidUniforms {
    float2 resolution;
    float2 touch;
    float2 touchVelocity;
    float touchActive;
    float dt;
    float viscosity;
    float diffusion;
    float time;
    float3 tint;
};

// MARK: - Hash / Noise

static float fr_hash(float2 p) {
    float3 p3 = fract(float3(p.xyx) * 0.1031);
    p3 += dot(p3, p3.yzx + 33.33);
    return fract((p3.x + p3.y) * p3.z);
}

static float fr_hash1(float n) {
    return fract(sin(n) * 43758.5453123);
}

static float fr_noise(float2 p) {
    float2 i = floor(p);
    float2 f = fract(p);
    f = f * f * (3.0 - 2.0 * f);
    float a = fr_hash(i);
    float b = fr_hash(i + float2(1, 0));
    float c = fr_hash(i + float2(0, 1));
    float d = fr_hash(i + float2(1, 1));
    return mix(mix(a, b, f.x), mix(c, d, f.x), f.y);
}

static float fr_fbm(float2 p, int octaves) {
    float v = 0.0;
    float a = 0.5;
    for (int i = 0; i < octaves; i++) {
        v += a * fr_noise(p);
        p *= 2.07;
        a *= 0.5;
    }
    return v;
}

// MARK: - Tree silhouette

// Returns mask in [0,1] for a vertical tree trunk centered at cx, with given width and bend.
// uv: flipped UV (y up)
static float trunkMask(float2 uv, float cx, float w, float bend, float t) {
    // Trunk slightly sways with wind
    float sway = sin(t * 0.4 + cx * 7.0) * 0.005;
    float dx = uv.x - (cx + bend * uv.y * uv.y + sway);
    // Trunk thickens slightly toward the base
    float width = w * mix(1.4, 0.7, uv.y);
    float m = smoothstep(width, width * 0.55, abs(dx));
    // Cut top (where canopy hides it)
    m *= smoothstep(0.95, 0.55, uv.y);
    // Fade into ground
    m *= smoothstep(0.0, 0.05, uv.y);
    return m;
}

// MARK: - Canopy (foliage layered fbm)

static float canopyDensity(float2 uv, float t, float scale, float drift) {
    float2 p = uv * scale + float2(t * drift * 0.10, t * drift * 0.04);
    float n = fr_fbm(p, 5);
    // Biased toward upper portion of frame
    float vBias = smoothstep(0.10, 0.85, uv.y);
    return n * vBias;
}

// Wind perturbation field, used to rustle the canopy and shift god rays
static float2 windField(float2 uv, float t) {
    float w1 = fr_noise(uv * 1.7 + float2(t * 0.25, 0.0));
    float w2 = fr_noise(uv * 0.7 + float2(0.0, t * 0.18));
    return float2(w1 - 0.5, w2 - 0.5);
}

// MARK: - Forest floor (leaf litter)

static float3 leafLitter(float2 uv, float t) {
    // Multi-scale noise with warm browns and greens
    float2 p = uv * 12.0;
    float n1 = fr_fbm(p, 4);
    float n2 = fr_fbm(p * 2.3 + 5.7, 3);

    // Per-leaf color variation
    float leafSeed = fr_hash(floor(p * 1.5));
    float3 leafCool   = float3(0.18, 0.22, 0.10); // damp moss-green
    float3 leafWarm   = float3(0.42, 0.26, 0.10); // dry oak leaf
    float3 leafDark   = float3(0.10, 0.07, 0.04); // shadow / soil
    float3 leafGold   = float3(0.55, 0.40, 0.14); // sunlit fallen leaf

    float3 col = mix(leafDark, leafCool, smoothstep(0.30, 0.55, n1));
    col = mix(col, leafWarm, smoothstep(0.50, 0.75, n2));
    col = mix(col, leafGold, smoothstep(0.78, 0.95, n2 * leafSeed));

    return col;
}

// MARK: - Bark texture

static float3 barkColor(float2 uv, float cx) {
    float2 p = float2((uv.x - cx) * 60.0, uv.y * 8.0);
    float ridges = fr_fbm(p, 4);
    float darkLines = sin(uv.y * 80.0 + ridges * 6.0) * 0.5 + 0.5;
    float3 base = float3(0.16, 0.11, 0.07);
    float3 high = float3(0.32, 0.22, 0.13);
    float3 col = mix(base, high, ridges);
    col *= mix(0.7, 1.0, darkLines);
    return col;
}

// MARK: - Forest render kernel

kernel void forest_render(
    texture2d<float, access::write> output [[texture(0)]],
    constant FluidUniforms &u             [[buffer(0)]],
    uint2 gid                             [[thread_position_in_grid]]
) {
    int w = int(u.resolution.x);
    int h = int(u.resolution.y);
    if (int(gid.x) >= w || int(gid.y) >= h) return;

    float2 uv = float2(gid) / u.resolution;
    // Flip Y: 0 = bottom (forest floor), 1 = top (canopy)
    float2 fUV = float2(uv.x, 1.0 - uv.y);

    float t = u.time;

    // ---- Touch mapping ----
    float2 touchUV = float2(u.touch.x / u.resolution.x, 1.0 - u.touch.y / u.resolution.y);
    float touchActive = (u.touchActive > 0.5) ? 1.0 : 0.0;

    // ---- Wind field ----
    float2 wind = windField(fUV, t);

    // ---- Background gradient ----
    // Deep forest: dark olive-green near top of canopy, warm earthy at floor
    float3 skyDeep   = float3(0.04, 0.07, 0.05); // canopy depth
    float3 midForest = float3(0.07, 0.10, 0.07);
    float3 floorWarm = float3(0.10, 0.08, 0.05);
    float3 bg = mix(floorWarm, midForest, smoothstep(0.10, 0.55, fUV.y));
    bg = mix(bg, skyDeep, smoothstep(0.55, 0.95, fUV.y));

    float3 color = bg;

    // ---- Forest floor (bottom 25%) ----
    if (fUV.y < 0.28) {
        // Perspective squash: floor stretches with distance
        float floorY = fUV.y / 0.28;          // 0 at horizon, 1 at near
        float persp = mix(0.25, 1.0, floorY); // sample density toward near
        float2 floorUV = float2(fUV.x, floorY * persp);

        float3 litter = leafLitter(floorUV, t);

        // Slight per-leaf wind-rustle: micro-displace by wind field
        float2 dispUV = floorUV + wind * 0.005;
        float3 litterB = leafLitter(dispUV, t);
        litter = mix(litter, litterB, 0.4);

        // Distance darkening (fog falls onto far floor)
        float dist = 1.0 - floorY;
        litter *= mix(1.0, 0.45, dist);

        // Touch press: brighten leaves under finger as if they're being disturbed
        if (touchActive > 0.5) {
            float d = length(fUV - touchUV);
            float pressGlow = exp(-d * d / 0.012) * 0.45;
            litter += float3(0.30, 0.22, 0.10) * pressGlow;
            // Shift hue toward warm under press
            litter = mix(litter, litter * float3(1.15, 1.0, 0.85), pressGlow);
        }

        // Blend floor over background
        float floorMix = smoothstep(0.28, 0.05, fUV.y);
        color = mix(color, litter, floorMix);
    }

    // ---- Background tree silhouettes (far layer) ----
    // 5 thin distant trunks, dark and slightly blurred
    {
        float farTrunks = 0.0;
        for (int i = 0; i < 5; i++) {
            float seed = float(i) * 11.7;
            float cx = fr_hash1(seed) * 1.0;
            float bend = (fr_hash1(seed + 1.0) - 0.5) * 0.05;
            farTrunks = max(farTrunks, trunkMask(fUV, cx, 0.005, bend, t));
        }
        // Soft and dark
        color = mix(color, float3(0.03, 0.04, 0.03), farTrunks * 0.7);
    }

    // ---- Mid-ground tree trunks (3 medium) ----
    {
        // Anchored positions
        float positions[3] = { 0.18, 0.62, 0.88 };
        float widths[3]    = { 0.018, 0.022, 0.016 };
        float bends[3]     = { 0.01, -0.015, 0.005 };

        for (int i = 0; i < 3; i++) {
            float cx = positions[i];
            float w0 = widths[i];
            float b  = bends[i];
            float m = trunkMask(fUV, cx, w0, b, t);
            if (m > 0.01) {
                float3 bark = barkColor(fUV, cx);
                color = mix(color, bark, m * 0.92);
            }
        }
    }

    // ---- Foreground hero trunk (left center, wider) ----
    {
        float cx = 0.36;
        float w0 = 0.045;
        float b  = -0.02;
        float m = trunkMask(fUV, cx, w0, b, t);
        if (m > 0.01) {
            float3 bark = barkColor(fUV, cx);
            // Add a vertical highlight on the right side (where light hits)
            float highlightX = (fUV.x - cx) / w0;
            float highlight = smoothstep(0.0, 0.7, highlightX) * smoothstep(1.5, 0.7, highlightX);
            bark += float3(0.18, 0.13, 0.05) * highlight * 0.6;
            color = mix(color, bark, m * 0.96);
        }
    }

    // ---- Canopy / leaves (top 80%) ----
    {
        // Near canopy (large leaves, lower contrast)
        float2 canopyUV = fUV + wind * 0.02;
        float densityNear = canopyDensity(canopyUV, t, 5.5, 1.0);
        float densityFar  = canopyDensity(canopyUV * 1.4 + float2(3.7, 9.1), t, 11.0, 0.6);

        // Combine: far filters near
        float density = densityNear * 0.65 + densityFar * 0.45;

        // Mask: only above the floor band
        float canopyMask = smoothstep(0.18, 0.45, fUV.y) * smoothstep(0.0, 0.5, density);

        // Leaf colors
        float3 leafShadow = float3(0.04, 0.07, 0.03);
        float3 leafMid    = float3(0.09, 0.15, 0.06);
        float3 leafLit    = float3(0.22, 0.32, 0.10);

        float3 leafCol = mix(leafShadow, leafMid, smoothstep(0.30, 0.55, density));
        leafCol = mix(leafCol, leafLit, smoothstep(0.62, 0.85, density));

        color = mix(color, leafCol, canopyMask);
    }

    // ---- God rays (komorebi) ----
    // Diagonal shafts of light from upper-left, broken up by canopy.
    {
        // Sun direction in screen space
        float2 sunPos = float2(0.18, 0.95);
        float2 toSun = sunPos - fUV;
        float distFromSun = length(toSun);

        // Project along ray direction
        float2 rayDir = normalize(float2(0.3, -1.0)); // downward-right
        // Distance perpendicular to ray
        float along = dot(fUV - sunPos, rayDir);
        float2 perp = (fUV - sunPos) - rayDir * along;
        float perpDist = length(perp);

        // Shaft pattern: along ray, modulated by horizontal noise
        float shaftBand = fr_noise(float2(perpDist * 18.0, along * 0.3) +
                                    float2(t * 0.2, 0.0));
        // Wind modulation: shafts subtly shift with wind field
        shaftBand += (wind.x * 0.15);

        float shafts = smoothstep(0.55, 0.85, shaftBand);

        // Falloff with distance from sun, and only above floor
        float distFalloff = exp(-distFromSun * 1.0);
        float floorFalloff = smoothstep(0.10, 0.40, fUV.y);

        // Volumetric mist amount: more visible where foliage is sparse
        float canopyGate = canopyDensity(fUV, t, 5.5, 1.0);
        float gate = 1.0 - smoothstep(0.55, 0.85, canopyGate);

        float rayIntensity = shafts * distFalloff * floorFalloff * gate * 0.55;

        // Warm sunlight color
        float3 sunWarm = float3(1.0, 0.85, 0.55);
        color += sunWarm * rayIntensity;

        // A soft direct sun bloom near the sun position (peeks through canopy)
        float sunGlow = exp(-distFromSun * distFromSun * 8.0) * 0.35 * gate;
        color += sunWarm * sunGlow;
    }

    // ---- Floating dust/pollen specks lit by sun ----
    {
        for (int i = 0; i < 18; i++) {
            float seed = float(i) * 51.317;
            float bx = fr_hash1(seed) * 0.9 + 0.05;
            float speed = fr_hash1(seed + 1.0) * 0.04 + 0.012;
            float lifetime = fr_hash1(seed + 2.0) * 6.0 + 4.0;
            float phase = fr_hash1(seed + 3.0);

            float cycleT = fract((t * speed / lifetime) + phase);
            // Slow downward drift, gentle horizontal sway
            float py = mix(0.85, 0.25, cycleT);
            float px = bx + sin(t * 0.3 + seed) * 0.04 + wind.x * 0.02;

            float fade = smoothstep(0.0, 0.15, cycleT) * smoothstep(1.0, 0.75, cycleT);

            float2 dustPos = float2(px, py);
            float d = length(fUV - dustPos);
            float size = 0.0025 + fr_hash1(seed + 4.0) * 0.0015;
            float dustGlow = exp(-d * d / (size * size));

            // Only visible in lit areas (near rays)
            float litness = smoothstep(0.7, 1.4, fUV.y * 0.5 + (1.0 - abs(px - 0.18)));
            color += float3(1.0, 0.92, 0.65) * dustGlow * fade * litness * 1.2;
        }
    }

    // ---- Touch ripple in canopy: rustles leaves outward ----
    if (touchActive > 0.5) {
        float d = length(fUV - touchUV);
        float ripple = sin(d * 60.0 - t * 8.0) * 0.5 + 0.5;
        float rippleStrength = exp(-d * 10.0) * ripple;
        // Brighten leaves slightly within ripple radius (above floor only)
        if (fUV.y > 0.20) {
            color += float3(0.10, 0.16, 0.06) * rippleStrength * 0.6;
        }
    }

    // ---- Atmospheric fog (depth cue) ----
    {
        float fogAmt = smoothstep(0.55, 0.95, fUV.y) * 0.18;
        float3 fogCol = float3(0.10, 0.13, 0.10);
        color = mix(color, fogCol, fogAmt);
    }

    // ---- Vignette ----
    float2 vigUV = uv - 0.5;
    float vignette = 1.0 - dot(vigUV, vigUV) * 1.1;
    color *= saturate(vignette);

    // ---- ACES tonemap ----
    color *= 1.25;
    float3 x = color;
    color = (x * (2.51 * x + 0.03)) / (x * (2.43 * x + 0.59) + 0.14);

    // Gamma
    color = pow(saturate(color), float3(1.0 / 2.2));

    // Tint
    color *= u.tint;

    output.write(float4(saturate(color), 1.0), gid);
}

import SwiftUI
import os.log

// MARK: - Touch state (shared reference between onTouch and VisualBody)

final class ForestTouchState: ObservableObject {
    @Published var touchLocation: CGPoint? = nil
    @Published var previousTouchLocation: CGPoint? = nil
    @Published var isTouching: Bool = false
}

// MARK: - Scene

/// Deep, ancient forest scene.
/// Visual: layered trunks, dense canopy, god rays (komorebi) cutting through
/// the leaves; atmospheric fog and slowly-drifting pollen specks.
/// Audio: low wind in branches + intermittent bird calls (registered via
/// `forest_ambient` asset; loaded if present in bundle).
/// Haptic: a single dry-leaf crunch on press, a multi-tap rustle on release,
/// and a low branch creak on long press.
struct ForestScene: SensoryScene {
    let id = "forest"
    let title = "Forest"

    // Shared reference: onTouch writes here, ForestVisual observes via @ObservedObject.
    private let touchState = ForestTouchState()

    private let logger = Logger(subsystem: "app.nagi", category: "ForestScene")

    @ViewBuilder
    func makeVisual(context: SensoryContext) -> some View {
        ForestVisual(touchState: touchState)
    }

    func prepare(context: SensoryContext) async {
        // Register haptic patterns (pre-loaded so first-fire latency stays inside the 30ms budget)
        do {
            let crunch = try HapticEngine.makeForestLeafCrunchPattern()
            context.haptics.register(name: "leaf_crunch", pattern: crunch)
        } catch {
            logger.warning("leaf_crunch pattern failed: \(error.localizedDescription, privacy: .public)")
        }

        do {
            let rustle = try HapticEngine.makeForestLeafRustlePattern()
            context.haptics.register(name: "leaf_rustle", pattern: rustle)
        } catch {
            logger.warning("leaf_rustle pattern failed: \(error.localizedDescription, privacy: .public)")
        }

        do {
            let creak = try HapticEngine.makeForestBranchCreakPattern()
            context.haptics.register(name: "branch_creak", pattern: creak)
        } catch {
            logger.warning("branch_creak pattern failed: \(error.localizedDescription, privacy: .public)")
        }

        // Register audio asset (graceful no-op if not yet bundled)
        guard let url = Bundle.main.url(forResource: "forest_ambient", withExtension: "wav")
                ?? Bundle.main.url(forResource: "forest_ambient", withExtension: "aiff")
                ?? Bundle.main.url(forResource: "forest_ambient", withExtension: "m4a") else {
            logger.warning("forest_ambient not found in bundle — skipping audio registration")
            return
        }
        context.audio.registerSoundAsset(identifier: "forest_ambient", url: url)
    }

    func onTouch(_ event: TouchEvent, context: SensoryContext) {
        switch event {
        case .began(let position):
            let eventID = context.clock.beginEvent(label: "forest.tap")
            touchState.previousTouchLocation = nil
            touchState.touchLocation = position
            touchState.isTouching = true
            context.haptics.play(name: "leaf_crunch")
            context.clock.completeEvent(id: eventID, actuator: "haptic")

        case .moved(let from, let to, _):
            touchState.previousTouchLocation = from
            touchState.touchLocation = to
            touchState.isTouching = true

        case .ended(let position):
            touchState.isTouching = false
            touchState.touchLocation = position
            context.haptics.play(name: "leaf_rustle")

        case .longPress(let position):
            // TODO: longPress detection — TouchReactor only emits began/moved/ended
            // via DragGesture. A LongPressGesture overlay in SceneStage or a
            // hold-duration timer in TouchReactor is required to generate this event.
            touchState.touchLocation = position
            context.haptics.play(name: "branch_creak")
        }
    }

    func teardown(context: SensoryContext) {
        // Engine lifecycle is owned by SceneDirector; nothing to do here.
    }
}

// MARK: - Visual body

private struct ForestVisual: View {
    @ObservedObject var touchState: ForestTouchState

    var body: some View {
        FluidMetalView(
            touchLocation: $touchState.touchLocation,
            previousTouchLocation: $touchState.previousTouchLocation,
            isTouching: $touchState.isTouching,
            textureType: .forest
        )
        .ignoresSafeArea()
    }
}
