+final class WavesTouchState: ObservableObject {

+// MARK: - Scene
+//
+// Theme: deep ocean at night with quiet moonlight rippling on the surface.
+// Each touch drops a soft moonlit ripple — a single short tap haptic at the
+// epicentre, then a `ripple` decreasing-tap pattern as the wave train fades.
+
+struct WavesScene: SensoryScene {
+    let id = "waves"
+    let title = "Waves"
+
+    // Shared reference: onTouch writes here, WavesVisual observes via @ObservedObject.
+    private let touchState = WavesTouchState()
+
+    private let logger = Logger(subsystem: "app.nagi", category: "WavesScene")
+
+    @ViewBuilder
+    func makeVisual(context: SensoryContext) -> some View {
+        WavesVisual(touchState: touchState)
+    }
+
+    func prepare(context: SensoryContext) async {
+        // Register haptic patterns (pre-load to stay within the 30ms SyncClock budget).
+        do {
+            let dropletTap = try HapticEngine.makeWavesDropletTapPattern()
+            context.haptics.register(name: "waves_droplet_tap", pattern: dropletTap)
+        } catch {
+            logger.warning("waves_droplet_tap pattern failed: \(error.localizedDescription, privacy: .public)")
+        }
+
+        do {
+            let rippleTrain = try HapticEngine.makeWavesRippleTrainPattern()
+            context.haptics.register(name: "waves_ripple_train", pattern: rippleTrain)
+        } catch {
+            logger.warning("waves_ripple_train pattern failed: \(error.localizedDescription, privacy: .public)")
+        }
+
+        do {
+            let deepSwell = try HapticEngine.makeWavesDeepSwellPattern()
+            context.haptics.register(name: "waves_deep_swell", pattern: deepSwell)
+        } catch {
+            logger.warning("waves_deep_swell pattern failed: \(error.localizedDescription, privacy: .public)")
+        }
+
+        // Register ambient audio asset (skip silently if the bundle has no wav).
+        guard let url = Bundle.main.url(forResource: "waves_ocean", withExtension: "wav")
+                ?? Bundle.main.url(forResource: "waves_ocean", withExtension: "aiff")
+                ?? Bundle.main.url(forResource: "waves_ocean", withExtension: "m4a") else {
+            logger.warning("waves_ocean not found in bundle — skipping audio registration")
+            return
+        }
+        context.audio.registerSoundAsset(identifier: "waves_ocean", url: url)
+    }
+
+    func onTouch(_ event: TouchEvent, context: SensoryContext) {
+        switch event {
+        case .began(let position):
+            let eventID = context.clock.beginEvent(label: "waves.tap")
+            touchState.previousTouchLocation = nil
+            touchState.touchLocation = position
+            touchState.isTouching = true
+            context.haptics.play(name: "waves_droplet_tap")
+            context.clock.completeEvent(id: eventID, actuator: "haptic")
+
+        case .moved(let from, let to, _):
+            touchState.previousTouchLocation = from
+            touchState.touchLocation = to
+            touchState.isTouching = true
+
+        case .ended(let position):
+            touchState.isTouching = false
+            touchState.touchLocation = position
+            context.haptics.play(name: "waves_ripple_train")
+
+        case .longPress(let position):
+            // TODO: longPress detection — TouchReactor only emits began/moved/ended
+            // via DragGesture. A LongPressGesture overlay in SceneStage or a
+            // hold-duration timer in TouchReactor is required to generate this event.
+            touchState.touchLocation = position
+            context.haptics.play(name: "waves_deep_swell")
+        }
+    }
+
+    func teardown(context: SensoryContext) {
+        // Engine lifecycle is owned by SceneDirector; nothing to do here.
+    }
+}
+
+// MARK: - Visual body

+private struct WavesVisual: View {

+    var body: some View {

+            previousTouchLocation: $touchState.previousTouchLocation,
+            isTouching: $touchState.isTouching,
+            textureType: .waves
+        )
+        .ignoresSafeArea()
+    }
+}
