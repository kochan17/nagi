+        case .moved(let from, let to, _):

+            touchState.touchLocation = to
+            touchState.isTouching = true

+        case .ended(let position):

+#include <metal_stdlib>
+using namespace metal;
+
+// Mirrors the layout in FluidShaders.metal — keep these fields in sync.
+struct FluidUniforms {
+    float2 resolution;
+    float2 touch;
+    float2 touchVelocity;
+    float touchActive;
+    float dt;
+    float viscosity;
+    float diffusion;
+    float time;
+    float3 tint;
+};
+
+// MARK: - Hash / noise utilities (mw_ prefix to avoid collisions across shader files)
+
+static float mw_hash(float2 p) {
+    float3 p3 = fract(float3(p.xyx) * 0.1031);
+    p3 += dot(p3, p3.yzx + 33.33);
+    return fract((p3.x + p3.y) * p3.z);
+}
+
+static float mw_hash1(float n) {
+    return fract(sin(n) * 43758.5453123);
+}
+
+static float mw_noise(float2 p) {
+    float2 i = floor(p);
+    float2 f = fract(p);
+    f = f * f * (3.0 - 2.0 * f);
+    float a = mw_hash(i);
+    float b = mw_hash(i + float2(1, 0));
+    float c = mw_hash(i + float2(0, 1));
+    float d = mw_hash(i + float2(1, 1));
+    return mix(mix(a, b, f.x), mix(c, d, f.x), f.y);
+}
+
+// FBM with slow drift — used for the calm sea surface texture
+static float mw_fbm(float2 p, float time, int octaves) {
+    float v = 0.0;
+    float a = 0.5;
+    float2 shift = float2(time * 0.04, time * 0.06);
+    for (int i = 0; i < octaves; i++) {
+        v += a * mw_noise(p + shift);
+        p = p * 2.05 + float2(1.7, 9.2);
+        a *= 0.5;
+        shift *= 1.15;
+    }
+    return v;
+}
+
+// MARK: - Wave height field
+// Sum of three travelling sine waves at different angles + low-frequency swell.
+// Returns elevation in roughly [-1, 1].
+static float mw_waveHeight(float2 p, float t) {
+    // Long-period swell rolling across the frame
+    float swell = sin(p.y * 2.4 + t * 0.35) * 0.5
+                + sin(p.y * 4.7 - t * 0.5 + p.x * 0.6) * 0.25;
+
+    // Crossing wind chop at slight angles for a non-rectilinear feel
+    float chop1 = sin((p.x * 1.1 + p.y * 0.3) * 7.0 + t * 1.2) * 0.18;
+    float chop2 = sin((p.x * 0.6 - p.y * 0.2) * 11.0 - t * 0.85) * 0.12;
+
+    // Fine-grain ripples (small sparkle on the surface)
+    float ripple = mw_noise(p * float2(20.0, 9.0) + float2(0.0, t * 0.5)) - 0.5;
+
+    return (swell + chop1 + chop2) * 0.55 + ripple * 0.18;
+}
+
+// Numerical derivative of wave height → surface normal
+static float3 mw_waveNormal(float2 p, float t) {
+    float e = 0.005;
+    float hL = mw_waveHeight(p - float2(e, 0.0), t);
+    float hR = mw_waveHeight(p + float2(e, 0.0), t);
+    float hD = mw_waveHeight(p - float2(0.0, e), t);
+    float hU = mw_waveHeight(p + float2(0.0, e), t);
+    float3 n = normalize(float3(hL - hR, hD - hU, 2.0 * e * 8.0));
+    return n;
+}

+// MARK: - moonlit_waves_render kernel
+
+kernel void moonlit_waves_render(
+    texture2d<float, access::write> output [[texture(0)]],
+    constant FluidUniforms &u             [[buffer(0)]],
+    uint2 gid                             [[thread_position_in_grid]]
+) {
+    int w = int(u.resolution.x);
+    int h = int(u.resolution.y);
+    if (int(gid.x) >= w || int(gid.y) >= h) return;
+
+    float2 uv = float2(gid) / u.resolution;
+    // Y increases downward in Metal textures; flip so 0 = bottom (near horizon-water interface), 1 = top (sky)
+    float2 fUV = float2(uv.x, 1.0 - uv.y);
+
+    float t = u.time;
+
+    // --- Touch in flipped UV space ---
+    float2 touchUV = float2(u.touch.x / u.resolution.x, 1.0 - u.touch.y / u.resolution.y);
+    float touchOn = (u.touchActive > 0.5) ? 1.0 : 0.0;
+
+    // ---- Sky (top portion above horizon) ----
+    // Horizon sits at fUV.y = 0.62
+    float horizon = 0.62;
+
+    // Deep night sky: indigo top fading to a slightly lighter band near the horizon
+    float skyT = smoothstep(horizon, 1.0, fUV.y);
+    float3 skyDeep    = float3(0.015, 0.020, 0.055);  // top of frame
+    float3 skyHorizon = float3(0.040, 0.055, 0.110);  // near horizon
+    float3 skyColor = mix(skyHorizon, skyDeep, skyT);
+
+    // Faint star field (only above horizon)
+    if (fUV.y > horizon) {
+        float2 starUV = fUV * float2(220.0, 220.0);
+        float starHash = mw_hash(floor(starUV));
+        float starMask = step(0.997, starHash);
+        // Twinkle
+        float twinkle = 0.6 + 0.4 * sin(t * (1.5 + starHash * 4.0) + starHash * 31.0);
+        skyColor += float3(0.85, 0.90, 1.0) * starMask * twinkle * 0.6;
+    }
+
+    // ---- Moon (high in the sky) ----
+    float2 moonCenter = float2(0.72, 0.86);
+    float moonRadius = 0.062;
+    float moonDist = length((fUV - moonCenter) * float2(u.resolution.x / u.resolution.y, 1.0));
+    float moonDisc = smoothstep(moonRadius, moonRadius * 0.92, moonDist);
+    // Soft outer halo
+    float moonHalo = exp(-moonDist * 6.0) * 0.5
+                   + exp(-moonDist * 18.0) * 0.4;
+    float3 moonColor = float3(0.95, 0.96, 1.0);
+    if (fUV.y > horizon) {
+        skyColor += moonColor * moonHalo * 0.35;
+    }
+    skyColor = mix(skyColor, moonColor * 1.6, moonDisc);
+
+    // ---- Sea (below horizon) ----
+    float3 color;
+    if (fUV.y < horizon) {
+        // Map sea region to a 2D plane that recedes toward the horizon.
+        // Perspective foreshortening: as fUV.y → horizon, the world Y stretches.
+        float seaT = fUV.y / horizon;                 // 0 at bottom, 1 at horizon
+        float perspective = 1.0 / max(0.06, 1.0 - seaT * 0.92);
+        float2 seaP = float2(fUV.x * perspective, (1.0 - seaT) * perspective);
+
+        // Wave height + normal at this point
+        float h = mw_waveHeight(seaP, t);
+        float3 n = mw_waveNormal(seaP, t);
+
+        // ---- Touch ripples: concentric expanding rings centered on touch ----
+        float rippleAdd = 0.0;
+        if (touchOn > 0.5) {
+            // Aspect-correct distance in UV
+            float2 dUV = (fUV - touchUV) * float2(u.resolution.x / u.resolution.y, 1.0);
+            float r = length(dUV);
+            // Up to 3 concentric rings, expanding outward
+            for (int k = 0; k < 3; k++) {
+                float age = float(k) * 0.18;
+                float ringR = 0.04 + age * 0.6;
+                float band = exp(-pow((r - ringR) / 0.012, 2.0));
+                rippleAdd += band * (0.65 - 0.18 * float(k));
+            }
+            // Tint the surface and lift the height slightly
+            h += rippleAdd * 0.25;
+        }
+
+        // ---- Base water color: deep midnight blue, darker in the foreground troughs ----
+        float depthShade = mix(1.0, 0.55, seaT);  // farther sea is lighter (atmospheric)
+        float3 deepWater = float3(0.010, 0.024, 0.075);
+        float3 nearWater = float3(0.020, 0.045, 0.115);
+        float3 water = mix(deepWater, nearWater, depthShade);
+        // Trough shading
+        float troughShade = smoothstep(-0.8, 0.6, h);
+        water *= mix(0.55, 1.05, troughShade);
+
+        // ---- Moonlight column reflection ----
+        // The bright stripe under the moon, broken up by the wave normals.
+        float2 moonOnSea = float2(moonCenter.x, horizon - 0.001);
+        // Aspect-correct horizontal distance to the column axis
+        float colX = abs(fUV.x - moonOnSea.x) * (u.resolution.x / u.resolution.y);
+        // Column gets wider closer to viewer
+        float colWidth = 0.10 + (1.0 - seaT) * 0.20;
+        float column = exp(-pow(colX / colWidth, 2.0));
+        // Specular highlight on wave facets pointing toward the moon
+        // Approximate light direction toward the moon (in screen space)
+        float2 toMoon = normalize(moonCenter - float2(fUV.x, horizon - 0.0001));
+        float facetDot = max(0.0, dot(n.xy, toMoon));
+        float specSharp = pow(facetDot, 6.0);
+        float spec = column * specSharp * mix(0.4, 1.4, 1.0 - seaT);
+
+        // Glittering sparkle on crests within the column
+        float crestMask = smoothstep(0.45, 0.85, h + 0.15);
+        float sparkleNoise = mw_noise(seaP * 80.0 + t * 1.2);
+        float sparkle = column * crestMask * smoothstep(0.78, 0.95, sparkleNoise);
+
+        // White caps on the steepest crests anywhere on the sea
+        float cap = smoothstep(0.85, 1.05, h) * mix(0.2, 0.6, depthShade);
+
+        color = water;
+        color += float3(0.78, 0.82, 1.0) * spec * 1.3;        // moonlit specular column
+        color += float3(0.95, 0.97, 1.0) * sparkle * 1.6;     // glints
+        color += float3(0.65, 0.72, 0.95) * cap * 0.45;       // foam caps
+
+        // ---- Touch ripple highlight: pale moon tint along the ring crests ----
+        if (touchOn > 0.5) {
+            color += float3(0.7, 0.78, 1.0) * rippleAdd * 0.45;
+        }
+
+        // ---- Soft mist near the horizon (atmospheric haze) ----
+        float mist = smoothstep(horizon - 0.10, horizon, fUV.y) * 0.55;
+        color = mix(color, float3(0.07, 0.09, 0.16), mist);
+    } else {
+        color = skyColor;
+    }
+
+    // ---- Horizon glow: a thin warmer band where moon column meets the sea ----
+    float horizonBand = exp(-pow((fUV.y - horizon) / 0.012, 2.0));
+    float horizonColX = abs(fUV.x - moonCenter.x) * (u.resolution.x / u.resolution.y);
+    float horizonGlow = horizonBand * exp(-pow(horizonColX / 0.18, 2.0));
+    color += float3(0.55, 0.62, 0.85) * horizonGlow * 0.55;
+
+    // ---- Subtle bloom around the moon ----
+    {
+        float bloom = exp(-moonDist * 12.0) * 0.4 + exp(-moonDist * 35.0) * 0.6;
+        color += moonColor * bloom * 0.18;
+    }
+
+    // ---- Vignette ----
+    float2 vigUV = uv - 0.5;
+    float vignette = 1.0 - dot(vigUV, vigUV) * 0.9;
+    color *= saturate(vignette);
+
+    // ---- ACES-ish tonemap (gentle, preserves the dark mood) ----
+    color *= 1.25;
+    float3 x = color;
+    color = (x * (2.51 * x + 0.03)) / (x * (2.43 * x + 0.59) + 0.14);
+
+    // Gamma
+    color = pow(saturate(color), float3(1.0 / 2.2));
+
+    // Tint (allow variant override; default tint is white)
+    color *= u.tint;
+
+    output.write(float4(saturate(color), 1.0), gid);
+}
