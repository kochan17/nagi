using namespace metal;

// Mirrors the layout in FluidShaders.metal — keep these fields in sync.
struct FluidUniforms {
    float2 resolution;
    float2 touch;
    float2 touchVelocity;
    float touchActive;
    float dt;
    float viscosity;
    float diffusion;
    float time;
    float3 tint;

// MARK: - Hash / Noise utilities (forest_ prefix avoids cross-file collisions)

static float forest_hash(float2 p) {
    float3 p3 = fract(float3(p.xyx) * 0.1031);
    p3 += dot(p3, p3.yzx + 33.33);
    return fract((p3.x + p3.y) * p3.z);
}

static float forest_hash1(float n) {
    return fract(sin(n) * 43758.5453123);
}

static float forest_noise(float2 p) {
    float2 i = floor(p);
    float2 f = fract(p);
    f = f * f * (3.0 - 2.0 * f);
    float a = forest_hash(i);
    float b = forest_hash(i + float2(1, 0));
    float c = forest_hash(i + float2(0, 1));
    float d = forest_hash(i + float2(1, 1));
    return mix(mix(a, b, f.x), mix(c, d, f.x), f.y);
}

// FBM for canopy / leaf clusters
static float forest_fbm(float2 p, float time, int octaves) {
    float value = 0.0;
    float amplitude = 0.5;
    // Slow, breeze-driven drift — leaves rustle horizontally
    float2 shift = float2(time * 0.18, time * 0.05);
    for (int i = 0; i < octaves; i++) {
        value += amplitude * forest_noise(p + shift);
        p *= 2.05;
        amplitude *= 0.5;
        shift *= 1.15;
    }
    return value;
}

// Slower FBM for fog / haze layers
static float forest_fog(float2 p, float time) {
    float value = 0.0;
    float amplitude = 0.5;
    float2 shift = float2(time * 0.04, time * 0.02);
    for (int i = 0; i < 3; i++) {
        value += amplitude * forest_noise(p + shift);
        p *= 2.0;
        amplitude *= 0.5;
        shift *= 1.1;
    }
    return value;
}

// MARK: - Trunk shape (vertical bands)
//
// Returns (mask, depth). depth in [0,1]: 0 = far, 1 = near.
static float2 forest_trunk(float2 fUV, float centerX, float halfWidth, float depth, float time) {
    // Slight organic wobble along Y so trunks aren't perfectly straight
    float wob = forest_noise(float2(fUV.y * 6.0, depth * 9.0)) - 0.5;
    float dx = abs(fUV.x - (centerX + wob * 0.012));
    // Trunk softness: hard core, soft edges
    float mask = smoothstep(halfWidth, halfWidth * 0.55, dx);
    // Trunks taper slightly toward the top (perspective)
    float taper = smoothstep(1.05, -0.05, fUV.y);
    mask *= mix(0.55, 1.0, taper);
    return float2(mask, depth);
}

// MARK: - Leaf cluster (canopy patches)
//
// Returns alpha 0..1 for a leaf cluster centered at `center`, scale `s`.
static float forest_leaf_cluster(float2 fUV, float2 center, float s, float time, float seed) {
    // Per-cluster wind sway phase
    float swayPhase = seed * 6.2831;
    float sway = sin(time * 0.9 + swayPhase) * 0.012 + sin(time * 1.7 + swayPhase * 1.7) * 0.006;
    float2 p = (fUV - center - float2(sway, 0.0)) / s;
    // Roughly elliptical falloff
    float r = length(p * float2(1.0, 1.4));
    float blob = smoothstep(1.0, 0.4, r);
    // Foliage texture inside the blob
    float leafy = forest_fbm(p * 5.0 + float2(seed * 13.0, seed * 7.0), time + seed, 4);
    leafy = smoothstep(0.45, 0.85, leafy);
    return blob * mix(0.55, 1.0, leafy);
}

                float3 lightBark = float3(0.13, 0.10, 0.07);
                float3 trunkCol = mix(darkBark, lightBark, bark * 0.7 + 0.15);
                // Mossy hint on shaded side
                float mossy = smoothstep(0.4, 0.0, fUV.y) * 0.4;
                trunkCol = mix(trunkCol, float3(0.06, 0.14, 0.07), mossy * 0.35);
                color = mix(color, trunkCol, td.x);
            }
        }
    }

    // ---------------- Foreground trunk (near, large, off-center) ----------------
    {
        float2 td = forest_trunk(fUV, 0.08, 0.045, 0.85, t);
        if (td.x > 0.001) {
            float bark = forest_noise(float2((fUV.x - 0.08) * 40.0, fUV.y * 6.0)) * 0.5
                       + forest_noise(float2((fUV.x - 0.08) * 9.0, fUV.y * 18.0)) * 0.5;
            float3 darkBark  = float3(0.025, 0.018, 0.012);
            float3 lightBark = float3(0.10, 0.075, 0.05);
            float3 trunkCol = mix(darkBark, lightBark, bark * 0.6 + 0.1);
            color = mix(color, trunkCol, td.x);
        }
    }

    // ---------------- Canopy: leaf clusters ----------------
    // Distribute many leaf blobs across the upper half
    {
        const int N = 14;
        for (int i = 0; i < N; i++) {
            float seed = float(i) * 31.7 + 0.13;
            float cx = forest_hash1(seed) * 1.1 - 0.05;
            float cy = 0.55 + forest_hash1(seed + 1.0) * 0.5;
            float s  = 0.10 + forest_hash1(seed + 2.0) * 0.12;
            float depthHint = forest_hash1(seed + 3.0);
            float a = forest_leaf_cluster(fUV, float2(cx, cy), s, t, seed);
            if (a > 0.001) {
                // Color varies between deep emerald and warm yellow-green
                float3 leafCold = float3(0.04, 0.14, 0.06);
                float3 leafWarm = float3(0.22, 0.34, 0.10);
                float3 leafCol = mix(leafCold, leafWarm, depthHint);
                // Darker at the underside (lower fUV.y inside the cluster)
                float undershade = smoothstep(0.95, 0.55, fUV.y);
                leafCol *= mix(0.55, 1.0, 1.0 - undershade);
                color = mix(color, leafCol, a * 0.85);
            }
        }
    }

    // ---------------- God rays (dappled sunlight through canopy) ----------------
    // Volumetric-feeling shafts: angled from upper-right toward lower-left.
    {
        // Project each pixel onto a direction perpendicular to ray axis.
        // Ray axis tilted ~22.5° from vertical.
        float2 rayDir = normalize(float2(-0.45, 1.0));
        float2 perpDir = float2(rayDir.y, -rayDir.x);
        // Coordinate along the perpendicular (so stripes form along this axis)
        float s = dot(fUV - float2(0.7, 0.95), perpDir);
        // Slow drift across the perpendicular axis (gentle wind in canopy)
        float drift = sin(t * 0.18) * 0.02;
        // Mask of bright stripes — multi-frequency to feel like leaf-broken light
        float ray = 0.0;
        ray += pow(0.5 + 0.5 * sin(s * 28.0 + drift * 30.0 + t * 0.25), 3.0) * 0.55;
        ray += pow(0.5 + 0.5 * sin(s * 11.0 + drift * 12.0 - t * 0.18), 4.0) * 0.40;
        ray += pow(0.5 + 0.5 * sin(s * 6.5  - t * 0.10), 6.0) * 0.30;
        // Modulate by canopy noise so rays appear to be filtered by leaves
        float canopyMask = forest_fbm(fUV * 5.0 + float2(0.3, 0.7), t * 0.15, 3);
        ray *= smoothstep(0.35, 0.75, canopyMask);
        // Falloff: rays brightest near top-right, fading to lower-left
        float along = dot(fUV - float2(0.0, 0.0), rayDir);
        float falloff = smoothstep(0.0, 1.6, along) * smoothstep(1.6, 0.4, along);
        // Horizontal limit: hide rays where they'd hit the deep forest floor
        float vertFalloff = smoothstep(0.05, 0.25, fUV.y);
        ray *= falloff * vertFalloff * 0.55;

        float3 sunWarm = float3(1.0, 0.92, 0.72);
        color += sunWarm * ray;
    }

    // ---------------- Soft ground fog ----------------
    {
        float fogMask = smoothstep(0.55, 0.0, fUV.y);
        if (fogMask > 0.001) {
            float fog = forest_fog(fUV * float2(3.0, 8.0) + float2(t * 0.05, 0.0), t);
            float fogAlpha = fogMask * smoothstep(0.4, 0.7, fog) * 0.35;
            float3 fogColor = float3(0.20, 0.24, 0.20);
            color = mix(color, fogColor, fogAlpha);
        }
    }

    // ---------------- Floating dust motes (illuminated where rays land) ----------------
    {
        for (int i = 0; i < 18; i++) {
            float seed = float(i) * 71.3 + 5.0;
            float speed = 0.04 + forest_hash1(seed + 1.0) * 0.05;
            float lifetime = 4.0 + forest_hash1(seed + 2.0) * 4.0;
            float phase = forest_hash1(seed + 3.0);
            float cycleT = fract((t / lifetime) + phase);

            // Drift down-left slowly (mirrors ray angle)
            float bx = forest_hash1(seed) * 1.1 - 0.05;
            float by = 0.95 - cycleT * 0.95;
            // Tiny lateral wobble
            bx += sin(t * 0.5 + seed) * 0.005;

            float2 motePos = float2(bx, by);
            float dist = length(fUV - motePos);
            float moteSize = 0.0018 + forest_hash1(seed + 4.0) * 0.0015;
            float glow = exp(-dist * dist / (moteSize * moteSize));
            // Only visible inside the bright ray cone (approximate)
            float2 rayDir = normalize(float2(-0.45, 1.0));
            float2 perpDir = float2(rayDir.y, -rayDir.x);
            float s2 = dot(motePos - float2(0.7, 0.95), perpDir);
            float rayHit = pow(0.5 + 0.5 * sin(s2 * 11.0), 4.0);
            float fadeIn = smoothstep(0.0, 0.1, cycleT) * smoothstep(1.0, 0.7, cycleT);
            color += float3(1.0, 0.92, 0.7) * glow * rayHit * fadeIn * 0.7;
        }
    }

    // ---------------- Touch interaction ----------------
    // A touch "stirs the underbrush" — a soft warm bloom + a small ripple
    // of brighter leaf rustle around the touch point.
    if (touchOn > 0.5) {
        float distToTouch = length(fUV - touchUV);
        // Warm bloom (like flashlight)
        float bloom = exp(-distToTouch * distToTouch / 0.020) * 0.55;
        color += float3(0.95, 0.78, 0.45) * bloom;
        // Local rustle: brighten leaf-noise contrast near touch
        float rustle = forest_fbm(fUV * 8.0 + t * 0.6, t, 3);
        float ringMask = smoothstep(0.18, 0.04, distToTouch) - smoothstep(0.04, 0.0, distToTouch);
        color += float3(0.10, 0.18, 0.06) * rustle * ringMask * 1.4;
    }

    // ---------------- Vignette ----------------
    float2 vigUV = uv - 0.5;
    float vignette = 1.0 - dot(vigUV, vigUV) * 1.1;
    color *= saturate(vignette);

    // ---------------- ACES tonemap ----------------
    color *= 1.25;
    float3 x = color;
    color = (x * (2.51 * x + 0.03)) / (x * (2.43 * x + 0.59) + 0.14);

    // Gamma
    color = pow(saturate(color), float3(1.0 / 2.2));

    // Tint
    color *= u.tint;

    output.write(float4(saturate(color), 1.0), gid);
}
