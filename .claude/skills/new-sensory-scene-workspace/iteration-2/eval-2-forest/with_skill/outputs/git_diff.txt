struct ForestScene: SensoryScene {
    let id = "forest"
    let title = "Forest"

    // Shared reference: onTouch writes here, ForestVisual observes via @ObservedObject.
    // Created once per ForestScene value; struct copy semantics preserve the reference.
    private let touchState = ForestTouchState()

    private let logger = Logger(subsystem: "app.nagi", category: "ForestScene")

    @ViewBuilder
    func makeVisual(context: SensoryContext) -> some View {
        ForestVisual(touchState: touchState)
    }

    func prepare(context: SensoryContext) async {
        // Register haptic patterns: deep, ancient forest with crunching leaf litter underfoot.
        do {
            let leafCrunch = try HapticEngine.makeForestLeafCrunchPattern()
            context.haptics.register(name: "leaf_crunch", pattern: leafCrunch)
        } catch {
            logger.warning("leaf_crunch pattern failed: \(error.localizedDescription, privacy: .public)")
        }

        do {
            let twigSnap = try HapticEngine.makeForestTwigSnapPattern()
            context.haptics.register(name: "twig_snap", pattern: twigSnap)
        } catch {
            logger.warning("twig_snap pattern failed: \(error.localizedDescription, privacy: .public)")
        }

        do {
            let leafRustle = try HapticEngine.makeForestLeafRustlePattern()
            context.haptics.register(name: "leaf_rustle", pattern: leafRustle)
        } catch {
            logger.warning("leaf_rustle pattern failed: \(error.localizedDescription, privacy: .public)")
        }

        // Register audio asset (skip silently if not yet bundled)
        guard let url = Bundle.main.url(forResource: "forest_ambient", withExtension: "wav")
                ?? Bundle.main.url(forResource: "forest_ambient", withExtension: "aiff") else {
            logger.warning("forest_ambient not found in bundle — skipping audio registration")
            return
        }
        context.audio.registerSoundAsset(identifier: "forest_ambient", url: url)
    }

    func onTouch(_ event: TouchEvent, context: SensoryContext) {
        switch event {
        case .began(let position):
            let eventID = context.clock.beginEvent(label: "forest.tap")
            touchState.previousTouchLocation = nil
            touchState.touchLocation = position
            touchState.isTouching = true
            // First step: dry leaves crunch underfoot
            context.haptics.play(name: "leaf_crunch")
            context.clock.completeEvent(id: eventID, actuator: "haptic")

        case .moved(let from, let to, _):
            touchState.previousTouchLocation = from
            touchState.touchLocation = to
            touchState.isTouching = true

        case .ended(let position):
            touchState.isTouching = false
            touchState.touchLocation = position
            // Lift: leaves rustle and settle back
            context.haptics.play(name: "leaf_rustle")

        case .longPress(let position):
            // Press deeper into the forest floor — a hidden twig snaps
            touchState.touchLocation = position
            context.haptics.play(name: "twig_snap")
        }
    }

    func teardown(context: SensoryContext) {
        // Engine lifecycle is owned by SceneDirector; nothing to do here.
    }

#include <metal_stdlib>
using namespace metal;

// Mirrors FluidUniforms layout in FluidShaders.metal — keep these fields in sync.
struct FluidUniforms {
    float2 resolution;
    float2 touch;
    float2 touchVelocity;
    float touchActive;
    float dt;
    float viscosity;
    float diffusion;
    float time;
    float3 tint;
};

// MARK: - Hash / noise utilities (prefixed forest_ to avoid collisions across shader files)

static float forest_hash(float2 p) {
    float3 p3 = fract(float3(p.xyx) * 0.1031);
    p3 += dot(p3, p3.yzx + 33.33);
    return fract((p3.x + p3.y) * p3.z);
}

static float forest_hash1(float n) {
    return fract(sin(n) * 43758.5453123);
}

static float forest_noise(float2 p) {
    float2 i = floor(p);
    float2 f = fract(p);
    f = f * f * (3.0 - 2.0 * f);
    float a = forest_hash(i);
    float b = forest_hash(i + float2(1, 0));
    float c = forest_hash(i + float2(0, 1));
    float d = forest_hash(i + float2(1, 1));
    return mix(mix(a, b, f.x), mix(c, d, f.x), f.y);
}

static float forest_fbm(float2 p, float time, int octaves) {
    float v = 0.0;
    float a = 0.5;
    float2 shift = float2(time * 0.05, -time * 0.03);
    for (int i = 0; i < octaves; i++) {
        v += a * forest_noise(p + shift);
        p *= 2.07;
        a *= 0.5;
        shift *= 1.15;
    }
    return v;
}

// Slow leaf-canopy FBM: drifts gently with breeze.
static float forest_canopy_fbm(float2 p, float time) {
    float v = 0.0;
    float a = 0.55;
    // Wind bows the canopy laterally
    float2 wind = float2(sin(time * 0.45) * 0.12 + time * 0.04,
                         cos(time * 0.32) * 0.08);
    for (int i = 0; i < 5; i++) {
        v += a * forest_noise(p + wind);
        p *= 1.95;
        a *= 0.52;
        wind *= 1.18;
    }
    return v;
}

// Voronoi-ish leaf pattern for canopy gaps.
static float forest_leaf_cells(float2 p, float time) {
    float2 i = floor(p);
    float2 f = fract(p);
    float minDist = 1.0;
    for (int yo = -1; yo <= 1; yo++) {
        for (int xo = -1; xo <= 1; xo++) {
            float2 g = float2(float(xo), float(yo));
            float2 o = float2(forest_hash(i + g + 17.0),
                              forest_hash(i + g + 91.0));
            // Animate jitter softly so gaps shimmer with breeze
            o = 0.5 + 0.5 * sin(time * 0.6 + 6.2831 * o);
            float2 r = g + o - f;
            float d = dot(r, r);
            minDist = min(minDist, d);
        }
    }
    return sqrt(minDist);
}

// MARK: - Tree silhouette helpers

// Returns log-style trunk mask + tinted bark color for a vertical tree
// centered at cx, with thickness `thick` (in UV units).
static float forest_trunk_mask(float fxNorm, float thick, float taper, float fy) {
    // Slight taper toward top
    float t = thick * (1.0 - taper * fy);
    return smoothstep(t, t * 0.6, abs(fxNorm));
}

// Bark grain — vertical streaks with low-frequency knots.
static float forest_bark(float2 uv, float seed) {
    float vert = forest_fbm(float2(uv.x * 14.0 + seed, uv.y * 1.5), 0.0, 3);
    float knots = forest_fbm(uv * float2(2.5, 1.2) + seed, 0.0, 3);
    return mix(vert, knots, 0.35);
}

                float moss = forest_fbm(float2((fUV.x - cx) * 10.0, fUV.y * 3.0), t * 0.02, 3);
                moss = smoothstep(0.45, 0.7, moss) * (1.0 - smoothstep(0.5, 0.9, fUV.y));

                float3 trunkColor = mix(float3(0.06, 0.045, 0.030),
                                       float3(0.18, 0.13, 0.085),
                                       bark);
                // Moss patches on the lower trunk
                trunkColor = mix(trunkColor, float3(0.10, 0.18, 0.08), moss * 0.7);
                // Shadow from canopy darkens upper trunk
                trunkColor *= mix(1.0, 0.55, smoothstep(0.4, 0.85, fUV.y));

                color = mix(color, trunkColor, mask);
            }
        }
    }

    // ============================================================
    // 4. Forest floor — leaf litter and shadows
    // ============================================================
    if (fUV.y < 0.22) {
        float floorY = fUV.y / 0.22;
        // Leaf litter: small clustered patches with warm browns and damp greens.
        float litter = forest_fbm(float2(fUV.x * 14.0, floorY * 8.0), t * 0.03, 4);
        float patches = forest_fbm(float2(fUV.x * 4.0, floorY * 2.0), t * 0.01, 3);
        float3 floorBase = mix(float3(0.025, 0.030, 0.018),
                              float3(0.13, 0.085, 0.040),
                              litter);
        // Damp-mossy patches
        floorBase = mix(floorBase, float3(0.05, 0.10, 0.045), smoothstep(0.55, 0.75, patches) * 0.5);
        // Floor receives little light — darken near the camera (bottom of screen)
        float floorShadow = mix(0.4, 1.0, floorY);
        color = mix(color, floorBase * floorShadow, smoothstep(0.0, 0.2, 0.22 - fUV.y));
    }

    // ============================================================
    // 5. Canopy / leaves (top 60% of screen, layered with parallax)
    // ============================================================
    // Three depth layers: deep canopy (back), mid leaves, near foliage.
    if (fUV.y > 0.30) {
        // Back layer — densest, darkest
        float backLeaves = forest_canopy_fbm(fUV * float2(7.0, 5.0) + float2(3.1, t * 0.06), t);
        float backMask = smoothstep(0.30, 0.85, fUV.y);
        float3 backLeafCol = mix(float3(0.030, 0.060, 0.035),
                                float3(0.075, 0.140, 0.060),
                                backLeaves);
        color = mix(color, backLeafCol, backMask * smoothstep(0.45, 0.75, backLeaves) * 0.85);

        // Mid layer — voronoi cells make leaf-shaped clusters with gaps for light
        float cellD = forest_leaf_cells(fUV * float2(11.0, 8.0) + float2(t * 0.02, 0.0), t);
        float midLeaf = 1.0 - smoothstep(0.15, 0.55, cellD);
        float3 midLeafCol = mix(float3(0.060, 0.115, 0.055),
                                float3(0.145, 0.235, 0.105),
                                forest_canopy_fbm(fUV * 9.0, t));
        color = mix(color, midLeafCol, midLeaf * smoothstep(0.40, 0.95, fUV.y) * 0.85);

        // Near foliage — bright tips at the very top, with breeze sway
        float swayX = sin(t * 0.7 + fUV.x * 4.0) * 0.012;
        float2 nearUV = float2(fUV.x + swayX, fUV.y);
        float nearLeaves = forest_canopy_fbm(nearUV * float2(15.0, 11.0) + float2(0.0, t * 0.08), t);
        float nearMask = smoothstep(0.65, 1.0, fUV.y);
        float3 nearLeafCol = mix(float3(0.10, 0.18, 0.075),
                                 float3(0.22, 0.34, 0.13),
                                 nearLeaves);
        color = mix(color, nearLeafCol, nearMask * smoothstep(0.5, 0.85, nearLeaves) * 0.7);
    }

    // ============================================================
    // 6. Dappled light — god rays piercing the canopy
    // ============================================================
    {
        // Light source position (off-screen top-right)
        float2 sunDir = normalize(float2(0.4, 0.85));
        // Project current pixel along ray direction
        float rayCoord = dot(fUV - float2(0.5, 0.0), sunDir);
        // Modulate by a slow noise on the perpendicular axis to break into shafts.
        float2 perp = float2(-sunDir.y, sunDir.x);
        float perpCoord = dot(fUV, perp);
        float shafts = forest_fbm(float2(perpCoord * 16.0, t * 0.08), 0.0, 3);
        // Use leaf cells to gate the rays — light only passes through gaps
        float gap = smoothstep(0.32, 0.55, forest_leaf_cells(fUV * float2(8.0, 6.0) + float2(2.0, t * 0.03), t));
        float ray = smoothstep(0.45, 0.85, shafts) * gap;
        // Brightest at mid-height, fading toward floor and canopy top
        float verticalRay = smoothstep(0.10, 0.50, fUV.y) * smoothstep(0.95, 0.55, fUV.y);
        float rayStrength = ray * verticalRay * 1.2;
        // Warm morning sunlight color
        float3 sunColor = float3(1.05, 0.85, 0.55);
        color += sunColor * rayStrength * 0.55;

        // Pools of light on the forest floor where rays land
        if (fUV.y < 0.25) {
            float pool = smoothstep(0.55, 0.85, forest_noise(float2(fUV.x * 6.0 + t * 0.05, fUV.y * 8.0)));
            float poolGap = gap;
            color += sunColor * pool * poolGap * 0.35 * smoothstep(0.0, 0.18, fUV.y);
        }
    }

    // ============================================================
    // 7. Floating dust motes / suspended particles in the light shafts
    // ============================================================
    {
        for (int i = 0; i < 18; i++) {
            float seed = float(i) * 79.31;
            // Drift slowly upward and laterally
            float speed = forest_hash1(seed) * 0.05 + 0.02;
            float lifetime = forest_hash1(seed + 1.0) * 6.0 + 4.0;
            float phase = forest_hash1(seed + 2.0);
            float baseX = forest_hash1(seed + 3.0) * 0.7 + 0.15; // 0.15..0.85
            float cycle = fract(phase + t / lifetime);
            float age = cycle * lifetime;

            float drift = sin(age * 0.6 + seed) * 0.04;
            float px = baseX + drift;
            float py = 0.20 + cycle * 0.65; // rises across mid-canopy

            float fade = smoothstep(0.0, 0.2, cycle) * smoothstep(1.0, 0.7, cycle);
            // Only visible when in/near a light shaft
            float dustGap = smoothstep(0.30, 0.60, forest_leaf_cells(float2(px, py) * float2(8.0, 6.0) + float2(2.0, t * 0.03), t));

            if (py > 0.0 && py < 1.0 && fade > 0.0) {
                float dist = length(fUV - float2(px, py));
                float size = 0.0035 + forest_hash1(seed + 4.0) * 0.003;
                float glow = exp(-dist * dist / (size * size));
                color += float3(1.0, 0.92, 0.7) * glow * fade * dustGap * 0.6;
            }
        }
    }

    // ============================================================
    // 8. Touch interaction — finger brushes the canopy
    // ============================================================
    if (u.touchActive > 0.5) {
        // Local rustle: increase leaf brightness and add a subtle warm glow halo.
        float rustleNoise = forest_fbm(fUV * 30.0 + t * 0.8, 0.0, 3);
        float rustle = touchHalo * smoothstep(0.4, 0.7, rustleNoise);
        color += float3(0.18, 0.30, 0.13) * rustle * 1.3;

        // Faint warm bloom right at touch
        color += float3(0.9, 0.75, 0.5) * exp(-distToTouch * distToTouch / 0.004) * 0.45;
    }

    // ============================================================
    // 9. Atmospheric haze / depth fog across the whole image
    // ============================================================
    {
        // Slight blue-green mist that thickens toward the back (mid-distance)
        float fogDepth = smoothstep(0.10, 0.55, fUV.y) * smoothstep(0.95, 0.55, fUV.y);
        float3 fogColor = float3(0.07, 0.11, 0.085);
        color = mix(color, fogColor, fogDepth * 0.18);
    }

    // ============================================================
    // 10. Vignette + tonemap + tint
    // ============================================================
    float2 vigUV = uv - 0.5;
    float vignette = 1.0 - dot(vigUV, vigUV) * 1.05;
    color *= saturate(vignette);

    // Mild exposure boost for the deep greens
    color *= 1.25;

    // ACES-ish tonemap
    float3 x = color;
    color = (x * (2.51 * x + 0.03)) / (x * (2.43 * x + 0.59) + 0.14);

    // Gamma
    color = pow(saturate(color), float3(1.0 / 2.2));

    // Variant tint multiplier
    color *= u.tint;

    output.write(float4(saturate(color), 1.0), gid);
}
